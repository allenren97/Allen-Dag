"""Read the business intake spreadsheet into a list of normalized row dicts."""
from __future__ import annotations

from pathlib import Path
from typing import Any

import pandas as pd


REQUIRED_COLUMNS = (
    "business_line",
    "cadence",
    "engine",
    "connection_id_import",
    "connection_id_export",
    "database",
    "schema",
    "table",
)


def _normalize(value: Any) -> Any:
    """Trim strings and coerce blank / 'None' literals / NaN / NaT / NA to ``None``."""
    # Excel blanks often arrive as NaN/float nan; spreadsheets also use string "None".
    # Downstream YAML uses real None for omitted optional fields like predicate.
    try:
        if pd.isna(value):
            return None
    except (TypeError, ValueError):
        pass
    if isinstance(value, str):
        v = value.strip()
        if v == "" or v.lower() == "none":
            return None
        return v
    return value


def read_intake_rows(intake_path: Path) -> list[dict[str, Any]]:
    """
    Return the data rows of the first worksheet as a list of dicts keyed by
    header. Empty rows and rows missing ``business_line`` / ``cadence`` are
    skipped. Raises ``ValueError`` if a required column is missing.
    """
    df = pd.read_excel(intake_path, sheet_name=0, dtype=object)
    # dtype=object keeps cells as-is (no accidental float dates). Strip headers
    # so " schema " matches our REQUIRED_COLUMNS spelling.
    new_columns = []
    for c in df.columns:
        if c is not None:
            new_columns.append(str(c).strip())
        else:
            new_columns.append("")
    df.columns = new_columns

    missing = []
    for c in REQUIRED_COLUMNS:
        if c not in df.columns:
            missing.append(c)
    if missing:
        raise ValueError(
            f"Intake header is missing required column(s): {missing}"
        )

    rows: list[dict[str, Any]] = []
    for raw in df.to_dict(orient="records"):
        row = {}
        for key, value in raw.items():
            row[key] = _normalize(value)

        empty_row = True
        for v in row.values():
            if v is not None:
                empty_row = False
                break
        if empty_row:
            continue
        # business_line + cadence define dag_id — without both we can't place the row.
        if not row.get("business_line") or not row.get("cadence"):
            continue
        rows.append(row)
    return rows
